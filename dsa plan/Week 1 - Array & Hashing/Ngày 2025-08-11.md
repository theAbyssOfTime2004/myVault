```python
def encode(self, strs: List[str]) -> str:
	return ', '.join(f"{len(s)}#{s}) for s in strs

def decode(self,s: str) -> List[str]:
	res = []
	i = 0
	while i < len(s):
		j = s.find('#', i)
		length = int(s[i:j])
		string = s[j+1 : j+1+length]
		res.append(string)
		i = j + 1 + length
		if i < len(s) and s[i:i+2] == ", ":
			i += 2
	return res
```

### B√†i 10. Valid Parentheses (20)
```python
def isValid(self, s):
	mapping = {')' : '(', ']' : '[', '}' : '{'}
	stack = []
	for bracket in s:
		if bracket in mapping:    # l√† bracket ƒë√≥ng
			top = stack.pop() if stack else None
				if mapping[bracket] != top:
				return False
		else:                     # l√† bracket m·ªü
			stack.append(bracket)
	return not stack
	
```

### B√†i 11. Majority Element (169)

- c√°ch th·ªß c√¥ng v·ªõi for v√† t·∫°o flag `candidate`:
```python
def majorityElement(self, nums):
	candidate = None
	count = 0
	for num in nums:
		if count == 0:
			candidate = num
		if candidate == num:
			count += 1
		else: 
			count -= 1
	return candidate
```
