\begin{frame}{The problem}
A model must \textbf{solve a hard problem it currently fails} --- improving only from \emph{its own attempts} and a verifiable \emph{pass/fail} reward.
\vfill
Two obstacles:
\begin{itemize}
  \item \alert{Sparse signal:} the reward says only \emph{whether} an attempt passed, not \emph{what} was wrong --- hard to know which part to fix (credit assignment).
  \item \alert{No correct attempt to learn from:} on a truly hard problem the model \textbf{rarely produces any correct rollout} --- nothing to reinforce or imitate.
\end{itemize}
\vfill
$\Rightarrow$ \textbf{Flat-reward trap:} no correct attempt $\Rightarrow$ no usable gradient $\Rightarrow$ learning \alert{stalls on exactly the hardest problems} --- the ones we care about.
\vfill
\centering\emph{Goal: accelerate this \textbf{discovery} on complex code generation.}
\end{frame}

% 3
\begin{frame}{Background: SDPO and the gap}
\textbf{Setting --- test-time training (TTT):} fix one hard problem, let the model improve over a few inference-time steps, then reset; measure how fast it \textbf{discovers} a solution (discovery@k).
\vfill
\textbf{SDPO} (H\"ubotter et al.) attacks the \emph{sparse-signal} obstacle: turn the \emph{rich feedback} verifiable environments already produce (failing tests, runtime errors) into a \textbf{dense} signal.
\begin{itemize}
  \item The feedback-conditioned model is a \textbf{self-teacher}; its per-token distribution is distilled into the student.
  \item Denser credit assignment than GRPO: a $K$-dim target per token, not one scalar per sequence.
\end{itemize}
\vfill
\textbf{The gap:} SDPO is \alert{student-first} --- it distills the student's \emph{own failed rollout}. On hard problems little is correct to learn from $\Rightarrow$ it still hits the flat-reward trap. \textbf{$\leftarrow$ our target.}
\end{frame}

% 5
\begin{frame}{Teacher-first: the loss}
Per-token KL --- pull the \emph{student} toward the \emph{self-teacher}'s (context-conditioned) next-token distribution, summed over the filtered good pool $\mathcal{G}$:
\vfill
\[
\mathcal{L}_{\text{TF}}(\theta) = \sum_{y \in \mathcal{G}}\;\sum_{t=1}^{|y|}
\mathrm{KL}\!\Big(\underbrace{\pi_\theta(\cdot\,|\,x, y_{<t})}_{\text{student}} \;\big\|\;
\mathrm{stopgrad}\,\underbrace{\pi_\theta(\cdot\,|\,x, c, y_{<t})}_{\text{self-teacher}}\Big)
\]
\vfill
\begin{itemize}
  \item \textbf{stopgrad} on the teacher: only the student updates --- stops the teacher collapsing onto it and ignoring $c$.
  \item Gradient per logit: $\partial\mathcal{L}/\partial z_v = \pi_S(v) - \pi_T(v)$ --- raise every token the teacher trusts more than the student.
  \item vs.\ student-first SDPO: \emph{same form}, but sum over the \alert{correct $y_{\text{good}}$} instead of the student's \emph{failed} $y_{\text{student}}$.
\end{itemize}
\vfill
{\footnotesize Implemented as top-$K$ reverse KL ($K{=}20$); teacher $=$ same weights run under \texttt{no\_grad} (self-distillation).}
\end{frame}

% 6
\begin{frame}{Research objectives}
\begin{itemize}
  \item \textbf{RO-method:} Establish whether teacher-first beats student-first at discovering hard code solutions --- and whether it holds at \emph{matched compute}.
  \item \textbf{RO1:} Determine whether the \emph{reprompt-template} formulation of the feedback affects discovery.
  \item \textbf{RO2:} Measure whether distillation from rich context \emph{suppresses epistemic verbalization} on code.
  \item \textbf{RO3:} Characterize the \emph{structural \& domain limitations} going from procedural code to value-only math.
\end{itemize}
\end{frame}

% 7
\begin{frame}{Setup}
\begin{itemize}
  \item Model: \textbf{Qwen3-4B} (LoRA), thinking ON --- the \emph{same} model on both code (LiveCodeBench v6) and math (AIME 2026 pilot).
  \item Per problem: reset to base, \textbf{15 TTT steps}, evaluate student PRE vs POST.
  \item \textbf{8 frontier code problems $\times$ 6 seeds} (medium scale) $\Rightarrow$ Wilcoxon signed-rank test.
  \item Metrics: pass@16, discovery@k, total generations + GPU-time.
\end{itemize}
\end{frame}

\begin{frame}{RO1: the reprompt template shapes discovery}
The template sets \emph{what feedback the teacher sees}, so it directly changes the distillation target. We probe three points: \textbf{T1} (minimal), \textbf{T2} (anchor), \textbf{T5} (reasoning-inducing).
\vfill
\begin{columns}
\column{0.44\textwidth}
\centering
\begin{tabular}{lcc}
\toprule
Template & idx12 & idx39 \\
         & (easy) & (hard) \\
\midrule
T1 minimal   & 0.69 & 0.04 \\
T2 anchor    & 0.94 & 0.07 \\
T5 reasoning & 0.95 & \alert{0.12} \\
\bottomrule
\end{tabular}\\[0.4em]
{\scriptsize POST pass@16, 6 seeds}
\column{0.56\textwidth}
\begin{itemize}
  \item On hard problems: \alert{T5 $>$ T2 $>$ T1} --- reasoning-inducing discovers best (significant).
  \item On easy problems the templates \textbf{saturate} (all near ceiling).
  \item Largest \textbf{marginal utility} when the reasoning template is paired with teacher-first on the hardest problems.
\end{itemize}
\end{columns}
\end{frame}

\begin{frame}{RO3: value vs.\ procedure --- when it works}
\begin{columns}
\column{0.58\textwidth}
\centering\includegraphics[width=\linewidth]{../figures/bc_fig5_boundary.pdf}
\column{0.42\textwidth}
\begin{itemize}
  \item Code output \emph{is} a method (a program that generalizes) $\Rightarrow$ distillation installs a procedure $\Rightarrow$ \textbf{escape}.
  \item Answer-only math is a value $\Rightarrow$ teacher can only copy $\Rightarrow$ \textbf{no escape}.
  \item \emph{Same} Qwen3-4B on both $\Rightarrow$ not a weaker model; reference type is the \emph{primary} difference (budget also differs) --- $n{=}2$ pilot, read as \alert{directional}.
\end{itemize}

% 13
\begin{frame}{Contributions}
\begin{enumerate}
  \item A \textbf{teacher-first} organization of execution-guided self-distillation (clean filter, never distils a copy).
  \item Teacher-first \textbf{significantly accelerates discovery} and \textbf{escapes the flat-reward trap} --- compute-fair (RO-method).
  \item A \textbf{reprompt-template effect}, strongest on hard problems (RO1).
  \item A measured \textbf{epistemic-suppression} result on code (RO2).
  \item A \textbf{value-versus-procedure boundary} characterizing \emph{when} the method helps.
\end{enumerate}
\end{frame}

% 14
\begin{frame}{Limitations \& future directions}
\textbf{Honest bounds (\S6.2):}
\begin{itemize}
  \item One \textbf{4B} Qwen model at a personal compute scale --- a characterization, not a scaling law.
  \item Math pilot bottlenecked by \textbf{AIME}'s answer-only feedback (no worked solutions to derive from); $n{=}2$.
  \item A stronger base (\textbf{8B+}) is expected to \emph{narrow} the TF-vs-SF margin on reachable tasks.
  \item Epistemic result tied to one model / lexical marker set.
\end{itemize}
\vfill
\textbf{Future directions:} method-bearing math references (\textbf{MATH-500}); a same-domain \emph{reference-type ablation} to test the boundary causally; scale to \textbf{8B+} and a second code benchmark.
\end{frame}

% 15
\begin{frame}{Conclusion}
\begin{itemize}
  \item \textbf{Teacher-first execution-guided self-distillation} accelerates test-time discovery in complex code generation.
  \item It escapes the flat-reward trap, the advantage is \textbf{compute-fair}, and it shows a regime-dependent epistemic effect.
  \item The \textbf{value-versus-procedure boundary} explains \emph{when} it works: an in-reach method, not just an answer.
\end{itemize}
\end{frame}
